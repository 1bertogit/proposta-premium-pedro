        // Funções para navegação de abas
        function showTab(tabId) {
            // Esconde todas as abas
            const tabPanes = document.querySelectorAll('.tab-pane');
            tabPanes.forEach(pane => {
                pane.classList.remove('active');
            });
            
            // Remove a classe ativa de todos os botões
            const tabButtons = document.querySelectorAll('.tab-button');
            tabButtons.forEach(button => {
                button.classList.remove('active');
            });
            
            // Mostra a aba selecionada
            document.getElementById(tabId).classList.add('active');
            
            // Adiciona a classe ativa ao botão clicado
            event.currentTarget.classList.add('active');
            
            // Rola a página para o topo da aba
            window.scrollTo({
                top: document.querySelector('.tab-content').offsetTop - 80,
                behavior: 'smooth'
            });
        }
        
        // Funções para metodologia
        function showMethodologyContent(contentId) {
            // Esconde todos os conteúdos de metodologia

            methodologyContents.forEach(content => {

            // Adiciona a classe ativa à aba clicada
